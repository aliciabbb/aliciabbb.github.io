插值方法处理bayer模式传感器原始数据

发表于 2018-03-07 | 分类于 计算机视觉 | | 
拜耳色彩滤波阵列（Bayer Color Filter Array）是非常有名的彩色图片的数字采集格式。色彩滤波器的模式如下图所示，由一半的G，1/4的R，1/4的B组成。原始采集的图像是单通道的灰度图像，放大会有较强的颗粒感。使用插值方法可以将其转化为彩色图像，并得到平滑的效果。
![这里写图片描述](http://img.blog.csdn.net/20180307221759353?watermark/2/text/aHR0cDovL2Jsb2cuY3Nkbi5uZXQvYm9vbF9j/font/5a6L5L2T/fontsize/400/fill/I0JBQkFCMA==/dissolve/70/gravity/SouthEast)
原始阵列中，每个点都代表采集的G、B或者R其中的一个值，所以需要插值估算另外两个分量。常用的插值方法为均值法。 
#插值红蓝方法实现
R和B通过线性领域插值，但这有四种不同的分布，如下图所示：
![这里写图片描述](http://img.blog.csdn.net/20180307223025225?watermark/2/text/aHR0cDovL2Jsb2cuY3Nkbi5uZXQvYm9vbF9j/font/5a6L5L2T/fontsize/400/fill/I0JBQkFCMA==/dissolve/70/gravity/SouthEast)
在（a）与（b）中，R和B分别取领域的平均值。
#插值绿算法实现
在（c）与（d）中，取领域的4个B或R的均值作为中间像素的B值。
![这里写图片描述](http://img.blog.csdn.net/20180307223343786?watermark/2/text/aHR0cDovL2Jsb2cuY3Nkbi5uZXQvYm9vbF9j/font/5a6L5L2T/fontsize/400/fill/I0JBQkFCMA==/dissolve/70/gravity/SouthEast)
由于人眼对绿光反应最敏感，对紫光和红光则反应较弱，因此为了达到更好的画质，需要对G特殊照顾。在上述（c）与（d）中，扩展开来就是上图的（e）与（f）中间像素G的取值，者也有一定的算法要求，不同的算法效果上会有差异。经过相关的研究，（e）（f）中间像素G值的算法如下：
![这里写图片描述](http://img.blog.csdn.net/20180307223458685?watermark/2/text/aHR0cDovL2Jsb2cuY3Nkbi5uZXQvYm9vbF9j/font/5a6L5L2T/fontsize/400/fill/I0JBQkFCMA==/dissolve/70/gravity/SouthEast)![这里写图片描述](http://img.blog.csdn.net/20180307223529211?watermark/2/text/aHR0cDovL2Jsb2cuY3Nkbi5uZXQvYm9vbF9j/font/5a6L5L2T/fontsize/400/fill/I0JBQkFCMA==/dissolve/70/gravity/SouthEast)
#源代码
#include "stdafx.h"
#include<opencv2/opencv.hpp>
#include<opencv2/highgui/highgui.hpp>
#include<opencv2/imgproc/imgproc.hpp>
using namespace cv;

int _tmain(int argc, _TCHAR argv[])
{
Mat lisa_gray,lisa_rgb,lisa_graycopy;
lisa_gray = imread(“lisa_Bayer.bmp”, 2|4);
lisa_graycopy = lisa_gray;
//lisa_rgb.create(lisa_gray.rows, lisa_gray.cols, CV_8UC3);
imshow(“gray”, lisa_gray);
cvtColor(lisa_gray, lisa_rgb, CV_BayerGR2BGR);
//Mat lisa_rgb = Mat::zeros(lisa_gray.rows, lisa_gray.cols, CV_8UC3);
int row = lisa_graycopy.rows;
int col = lisa_graycopy.cols;
int colc = lisa_rgb.cols lisa_rgb.channels();
for (int i = 2; i < row-2; i++)
{
uchar data1 = lisa_gray.ptr(i);
uchar data11 = lisa_gray.ptr(i-1);
uchar data12 = lisa_gray.ptr(i+1);
uchar data13 = lisa_gray.ptr(i - 2);
uchar data14 = lisa_gray.ptr(i + 2);
uchar data2 = lisa_rgb.ptr(i);
//uchar data21 = lisa_rgb.ptr(i-1);
//uchar data22 = lisa_rgb.ptr(i+1);
for (int j = 6; j < colc-6; j++)
{
int k = (j+1) % 3;
int m = (j+1) / 3 + 1;
if (k == 0) m -= 1;
if ((i + 1) % 2 == 1 && m % 2 == 0 && j%3 == 0) //奇数行、偶数列 G
{
m = m - 1;
data2[j] = (data12[m] + data11[m]) / 2;
}
else if ((i + 1) % 2 == 1 && m % 2 == 0 && j % 3 == 1)
{
m = m - 1;
data2[j] = data1[m];
}
else if ((i + 1) % 2 == 1 && m % 2 == 0 && j % 3 == 2)
{
m = m - 1;
data2[j] = (data1[m + 1] + data1[m - 1]) / 2;
}
else if ((i + 1) % 2 == 1 && m % 2 == 1 && j % 3 == 0) //ji数行、ji数列 R
{
m = m - 1;
data2[j] = (data12[m + 1] + data12[m - 1] + data11[m +1 ] + data11[m - 1]) / 4;
}
else if ((i + 1) % 2 == 1 && m % 2 == 1 && j % 3 == 1)
{
m = m - 1;
//int m1 = abs(data13[m] - data14[m]);
//int n1 = abs(data1[m - 2] - data1[m + 2]);
//if (n1 > m1) data2[j] = (data11[m] + data12[m]) / 2;
//else if (n1 < m1) data2[j] = (data11[m] + data1[m - 1]) / 2;
data2[j] = (data1[m + 1] + data1[m - 1] + data11[m] + data12[m]) / 4;
}
else if ((i + 1) % 2 == 1 && m % 2 == 1 && j % 3 == 2)
{
m = m - 1;
data2[j] = data1[m];
}
else if ((i + 1) % 2 == 0 && m % 2 == 1 && j % 3 == 0) //偶数行、奇数列 G
{
m = m - 1;
data2[j] = (data1[m + 1] + data1[m - 1]) / 2;
}
else if ((i + 1) % 2 == 0 && m % 2 == 1 && j % 3 == 1)
{
m = m - 1;
data2[j] = data1[m];
}
else if ((i + 1) % 2 == 0 && m % 2 == 1 && j % 3 == 2)
{
m = m - 1;
data2[j] = (data12[m] + data11[m]) / 2;
}
else if ((i + 1) % 2 == 0 && m % 2 == 0 && j % 3 == 0) //o数行、o数列 B
{
m = m - 1;
data2[j] = data12[m];
}
else if ((i + 1) % 2 == 0 && m % 2 == 0 && j % 3 == 1)
{
m = m - 1;
//int m11 = abs(data13[m] - data14[m]);
//int n11 = abs(data1[m - 2] - data1[m + 2]);
//if (n11 > m11) data2[j] = (data11[m] + data12[m]) / 2;
//else if (n11 < m11) data2[j] = (data11[m] + data1[m - 1]) / 2;
data2[j] = (data1[m + 1] + data1[m - 1] + data11[m] + data12[m]) / 4;
}
else if ((i + 1) % 2 == 0 && m % 2 == 0 && j % 3 == 2)
{
m = m - 1;
data2[j] = (data12[m + 1] + data12[m - 1] + data11[m + 1] + data11[m - 1]) / 4;
}
}
}
imshow(“color”, lisa_rgb);
imwrite(“my.jpg”, lisa_rgb);
std::cout << col << std::endl;
waitKey(0);
return 0;
}

注意：不同的CCD可能阵列排列方式不同，自己要根据实际情况做出修改
